   

SELECT 
Patient.Pat_ID1, 
Last_Name, 
First_Name, 
Ident.IDA, 
Ident.IDB, 
Ident.IDC, 
Clin_Status

FROM Patient
INNER JOIN Ident ON Patient.Pat_ID1 = Ident.Pat_ID1 AND Ident.Version = 0
Where 1=1 
AND Clin_Status <>10
AND 
(UPPER(Last_Name) LIKE 'TEST%'
OR  UPPER(Last_Name) LIKE 'ZZ%'
OR  UPPER(Last_Name) LIKE 'XX%'
--OR NOT(Ident.IDA LIKE '[0-9]%')
--OR len(Ident.IDA) <> 7
)
